# deploy os servidores no modo swarm
docker stack deploy -c docker-compose.yml todoapi

# comando para subir as instâncias
docker-compose -f docker-compose.yml up

# comando para parar o stack - sem uso por enquanto
docker stack rm todoapi

# comando para parar todos os containers
docker stop $(docker ps -a -q)
docker rm $(docker ps -a -q)

# visualizar logs de um container específico que está rodando
docker logs --tail 50 --follow --timestamps #id

# rodar a instância docker
docker run -d -p 5000:5000 --name todoapi mateusleonardi/dotnetcore

# entrar no bash do docker
docker exec -it #id bash

# listar containers em execução
docker ps

# listar todas as imagens na sua máquina
docker images
